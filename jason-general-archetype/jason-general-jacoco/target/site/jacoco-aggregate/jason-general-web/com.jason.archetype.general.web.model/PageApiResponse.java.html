package com.jason.archetype.general.web.model;

import io.swagger.annotations.ApiModel;
import io.swagger.annotations.ApiModelProperty;

import java.util.List;

/**
 * @author jasonCQ
 * @version 1.0
 */
@ApiModel( description = "分页查询Api返回结果")
public class PageApiResponse<T> extends ApiResponse<List<T>> {
    @ApiModelProperty("总数")
    private long total;

    public PageApiResponse() {
    }

    public long getTotal() {
        return this.total;
    }

    public void setTotal(long total) {
        this.total = total;
    }

    public static <T> PageApiResponse<T> success(PageResultModel<T> pageResultModel) {
        PageApiResponse<T> pageApiResponse = new PageApiResponse();
        pageApiResponse.setState(true);
        pageApiResponse.setTotal(pageResultModel.getTotal());
        pageApiResponse.setData(pageResultModel.getDataList());
        return pageApiResponse;
    }
}

class PageResultModel<T> {
    private List<T> dataList;
    private long total;

    public List<T> getDataList() {
        return this.dataList;
    }

    public void setDataList(List<T> dataList) {
        this.dataList = dataList;
    }

    public long getTotal() {
        return this.total;
    }

    public void setTotal(long total) {
        this.total = total;
    }

    public PageResultModel(List<T> dataList, long total) {
        this.dataList = dataList;
        this.total = total;
    }
}
